        python virtual enviornment- python -m venv myenv

        list down python enviornment- you cannot list down python enviornment
        but Manually check python enviornment where you created them- dir /s /b venv
        activate python enviornment- myenv\Scripts\activate
        deactivate python enviornment- deactivate
        conda virtual enviornment- conda create -n myenv python=3.10
        delete conda virtual enviornment- conda remove --name myenv --all
        list down conda enviornment- conda info --envs
        activate conda enviornment- conda activate myenv
        deactivate conda enviornment- deactivate
